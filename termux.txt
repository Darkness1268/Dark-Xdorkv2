# Tool installer
install-tools() {
    echo -e "${FIRE1}Installing tools...${NC}"
    
    # Update first
    pkg update && pkg upgrade -y
    
    # Install essentials
    pkg install -y git python nodejs golang ruby perl nmap hydra sqlite curl wget
    
    # Python tools
    pip install requests beautifulsoup4 selenium colorama
    
    # Clone tools
    echo -e "${FIRE2}Cloning tools...${NC}"
    
    # SQLMap
    if [ ! -d ~/sqlmap-dev ]; then
        git clone https://github.com/sqlmapproject/sqlmap.git ~/sqlmap-dev
    fi
    
    # XSStrike
    if [ ! -d ~/XSStrike ]; then
        git clone https://github.com/s0md3v/XSStrike.git ~/XSStrike
    fi
    
    # Nuclei templates
    if [ ! -d ~/nuclei-templates ]; then
        git clone https://github.com/projectdiscovery/nuclei-templates.git ~/nuclei-templates
    fi
    
    echo -e "${FIRE3}Tools installed! Restart Termux.${NC}"
}
